js获取元素宽高

document.getElementById("border").style.width="" //js原生.style.width或者.width取不到值 1.元素未设置宽度值。 2.元素设置了宽度值，但设置在内联或外联样式表中，而非内嵌式的

document.getElementById("border").clientWidth=220 //获取content+padding的宽度

document.getElementById("border").offsetWidth=240 //获取content+padding+border的宽度

jquery获取元素宽高

$("#border").width()=200 //只是获取content宽度

$("#border").innerWidth()=220 //获取content+padding的宽度

$("#border").outerWidth()=240 //获取content+padding+border的宽度

$("#border").outerWidth(true)=260 //获取content+padding+border+margin的宽度
